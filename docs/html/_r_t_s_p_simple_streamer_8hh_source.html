359  // We handle only 4,8,16,20,24 bits-per-sample audio:
360  if (bitsPerSample % 4 != 0 || bitsPerSample < 4 || bitsPerSample > 24 ||
361  bitsPerSample == 12) {
362  *env << "The input file contains " << bitsPerSample
363  << " bit-per-sample audio, which we don't handle\n";
364  return false;
365  }
366 
367  unsigned bitsPerSecond = samplingFrequency * bitsPerSample * numChannels;
368 
369  *env << "Audio source parameters:\n\t" << samplingFrequency << " Hz, ";
370  *env << bitsPerSample << " bits-per-sample, ";
371  *env << numChannels << " channels => ";
372  *env << bitsPerSecond << " bits-per-second\n";
373 
374  char const* mimeType;
375  // by default, unless a static RTP payload type can be used
376  unsigned char payloadFormatCode = 96;
377 
378  // Add in any filter necessary to transform the data prior to streaming.
379  // (This is where any audio compression would get added.)
380  switch (audioFormat) {
381  case WA_PCM: {
382  if (bitsPerSample == 16) {
383  // Note that samples in the WAV audio file are in little-endian order.
384  // Add a filter that converts from little-endian to network
385  // (big-endian) order:
386  source = EndianSwap16::createNew(*env, pcmSource());
387  if (source == NULL) {
388  *env << "Unable to create a little->bit-endian order filter from "
389  "the PCM audio source: "
390  << env->getResultMsg() << "\n";
391  return false;
392  }
393  *env << "Converting to network byte order for streaming\n";
394  mimeType = "L16";
395  if (samplingFrequency == 44100 && numChannels == 2) {
396  payloadFormatCode = 10; // a static RTP payload type
397  } else if (samplingFrequency == 44100 && numChannels == 1) {
398  payloadFormatCode = 11; // a static RTP payload type
399  }
400  } else if (bitsPerSample == 20 || bitsPerSample == 24) {
401  // Add a filter that converts from little-endian to network
402  // (big-endian) order:
403  source = EndianSwap24::createNew(*env, pcmSource());
404  if (source == NULL) {
405  *env << "Unable to create a little->bit-endian order filter from "
406  "the PCM audio source: "
407  << env->getResultMsg() << "\n";
408  return false;
409  }
410  *env << "Converting to network byte order for streaming\n";
411  mimeType = bitsPerSample == 20 ? "L20" : "L24";
412  } else {
413  // bitsPerSample == 8 (we assume that bitsPerSample == 4 is
414  // only for WA_IMA_ADPCM)
415  // Don't do any transformation; send the 8-bit PCM data 'as is':
416  mimeType = "L8";
417  }
418  break;
419 
420  case WA_PCMU: {
421  mimeType = "PCMU";
422  if (samplingFrequency == 8000 && numChannels == 1) {
423  payloadFormatCode = 0; // a static RTP payload type
424  }
425  } break;
426 
427  case WA_PCMA: {
428  if (samplingFrequency == 8000 && numChannels == 1) {
429  payloadFormatCode = 8; // a static RTP payload type
430  }
431  } break;
432 
433  case WA_IMA_ADPCM: {
434  mimeType = "DVI4";
435  // Use a static payload type, if one is defined:
436  if (numChannels == 1) {
437  if (samplingFrequency == 8000) {
438  payloadFormatCode = 5; // a static RTP payload type
439  } else if (samplingFrequency == 16000) {
440  payloadFormatCode = 6; // a static RTP payload type
441  } else if (samplingFrequency == 11025) {
442  payloadFormatCode = 16; // a static RTP payload type
443  } else if (samplingFrequency == 22050) {
444  payloadFormatCode = 17; // a static RTP payload type
445  }
446  }
447  } break;
448 
449  default:
450  *env << "Unknown audio format code \"" << mimeType << "\" in WAV file header\n";
451  return false;
452  }
453  }
454 
455  *env << "Using mime: \"" << audioFormat << "\" \n";
456  sink = SimpleRTPSink::createNew(*env, rtpGroupsock, payloadFormatCode, samplingFrequency,
457  "audio", mimeType, numChannels);
458 
459  unsigned estimatedSessionBandwidth = (bitsPerSecond + 500)/1000;
460  rtcpInstance = RTCPInstance::createNew(*env, rtcpGroupsock,
461  estimatedSessionBandwidth, CNAME,
462  sink, NULL /* we're a server */,
463  cfg.isSSM /* we're a SSM source*/);
